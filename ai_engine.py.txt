import webbrowser
import os
import requests

# 👉 इथे तुझा OpenAI API key टाक
OPENAI_API_KEY = "YOUR_API_KEY_HERE"

def ask_ai(message):
    try:
        url = "https://api.openai.com/v1/chat/completions"

        headers = {
            "Authorization": f"Bearer {OPENAI_API_KEY}",
            "Content-Type": "application/json"
        }

        data = {
            "model": "gpt-4o-mini",
            "messages": [
                {"role": "system", "content": "You are a smart AI assistant like Jarvis."},
                {"role": "user", "content": message}
            ]
        }

        res = requests.post(url, headers=headers, json=data)
        result = res.json()

        return result["choices"][0]["message"]["content"]

    except:
        return "AI error bhava"

def process_command(message):
    msg = message.lower()

    # 🔥 AUTOMATION
    if "open youtube" in msg:
        webbrowser.open("https://youtube.com")
        return "Opening YouTube 🔥"

    if "open google" in msg:
        webbrowser.open("https://google.com")
        return "Opening Google 🚀"

    if "search" in msg:
        query = msg.replace("search", "")
        webbrowser.open(f"https://www.google.com/search?q={query}")
        return f"Searching {query}"

    if "open calculator" in msg:
        try:
            os.system("calc")
        except:
            pass
        return "Opening Calculator"

    # 🔥 AI BRAIN
    return ask_ai(message)
